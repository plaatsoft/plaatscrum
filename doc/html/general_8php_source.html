163 function t() {
164 
165  global $lang;
166 
167  $numArgs = func_num_args();
168 
169  $temp = $lang[func_get_arg(0)];
170 
171  $pos = 0;
172  $i = 1;
173 
174  while (($pos = strpos($temp, "%s", $pos)) !== false) {
175  if ($i >= $numArgs) {
176  throw new InvalidArgumentException("Not enough arguments passed.");
177  }
178 
179  $temp = substr($temp, 0, $pos) . func_get_arg($i) . substr($temp, $pos + 2);
180  $pos += strlen(func_get_arg($i));
181  $i++;
182  }
183 
184  $temp = mb_convert_encoding($temp, "UTF-8", "HTML-ENTITIES" );
185  return $temp;
186 }

194 function udate($format, $utimestamp = null) {
195  if (is_null($utimestamp)) {
196  $utimestamp = microtime(true);
197  }
198 
199  $timestamp = floor($utimestamp);
200  $milliseconds = round(($utimestamp - $timestamp) * 1000000);
201 
202  return date(preg_replace('`(?<!\\\\)u`', $milliseconds, $format), $timestamp);
203 }
