170 function plaatscrum_db_execute_sql_file($version) {
171 
172  $filename = 'database/patch-'.$version.'.sql';
173 
174  $commands = file_get_contents($filename);
175 
176  //delete comments
177  $lines = explode("\n",$commands);
178  $commands = '';
179  foreach($lines as $line){
180  $line = trim($line);
181  if( $line && !startsWith($line,'--') ){
182  $commands .= $line . "\n";
183  }
184  }
185 
186  //convert to array
187  $commands = explode(";\n", $commands);
188 
189  //run commands
190  $total = $success = 0;
191  foreach($commands as $command){
192  if(trim($command)){
193  $success += (@plaatscrum_db_query($command)==false ? 0 : 1);
194  $total += 1;
195  }
196  }
197 
198  //return number of successful queries and total number of queries found
199  return array(
200  "success" => $success,
201  "total" => $total
202  );
203 }
