154 function plaatscrum_export_do() {
155 
156  /* input */
157  global $user;
158  global $export_type;
159  global $export_sort;
160 
161  $query = 'select a.number, a.status, a.sprint_id, a.summary, a.description, b.name, a.points, a.reference,';
162  $query .= 'a.date, a.prio, a.type, c.number as sprint_number ';
163  $query .= 'from story a left join tuser b on a.user_id=b.user_id left join sprint c on a.sprint_id=c.sprint_id ';
164  $query .= 'where a.project_id='.$user->project_id.' and a.deleted=0 ';
165 
166  switch ($export_sort) {
167 
168  default:
169  $query .= 'order by a.number';
170  break;
171 
172  case 1:
173  $query .= 'order by a.status, a.number';
174  break;
175 
176  case 2:
177  $query .= 'order by a.points, a.number';
178  break;
179 
180  case 3:
181  $query .= 'order by a.prio, a.number';
182  break;
183 
184  case 4:
185  $query .= 'order by a.type, a.number';
186  break;
187 
188  case 5:
189  $query .= 'order by sprint_number, a.number';
190  break;
191  }
192 
193  $result = plaatscrum_db_query($query);
194 
195  switch ($export_type) {
196 
197  case 0:
198  plaatscrum_export_csv_do($result);
199  plaatscrum_info($user->name.' ['.$user->user_id.'] csv export project ['.$user->project_id.']');
200  break;
201 
202  case 1:
203  plaatscrum_export_xml_do($result);
204  plaatscrum_info($user->name.' ['.$user->user_id.'] xml export project ['.$user->project_id.']');

206 
207  case 2:
208  plaatscrum_export_json_do($result);
209  plaatscrum_info($user->name.' ['.$user->user_id.'] json export project ['.$user->project_id.']');
